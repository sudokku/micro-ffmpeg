**Step 1: Create `src/utils/computeTimestamps.ts`** — pure function, testable without WASM:

```typescript
/**
 * Compute frame extraction timestamps: 1 frame per 5 seconds, min 1 frame.
 * Timestamps are centered within each interval for representative frames.
 */
export function computeTimestamps(duration: number): number[] {
  const count = Math.max(1, Math.floor(duration / 5))
  return Array.from({ length: count }, (_, i) => (i + 0.5) * (duration / count))
}
```

**Step 2: Extend `src/workers/ffmpeg.worker.ts`** — add FFmpeg load + extractFrames:

```typescript
import * as Comlink from 'comlink'
import { FFmpeg } from '@ffmpeg/ffmpeg'
import { fetchFile } from '@ffmpeg/util'

let ffmpeg: FFmpeg | null = null

async function ensureLoaded(): Promise<FFmpeg> {
  if (!ffmpeg) {
    ffmpeg = new FFmpeg()
    await ffmpeg.load()
  }
  return ffmpeg
}

const api = {
  async ping(): Promise<'pong'> {
    return 'pong'
  },

  async extractFrames(fileData: Uint8Array, timestamps: number[]): Promise<Blob[]> {
    const ff = await ensureLoaded()
    await ff.writeFile('input', fileData)
    const blobs: Blob[] = []
    for (let i = 0; i < timestamps.length; i++) {
      const outName = `frame_${i}.jpg`
      await ff.exec([
        '-ss', String(timestamps[i]),
        '-i', 'input',
        '-vframes', '1',
        '-q:v', '5',
        outName,
      ])
      const data = await ff.readFile(outName) as Uint8Array
      blobs.push(new Blob([data], { type: 'image/jpeg' }))
      await ff.deleteFile(outName)
    }
    await ff.deleteFile('input')
    return blobs
  },
}

Comlink.expose(api)
export type FfmpegWorkerApi = typeof api
```

NOTE: The worker accepts `Uint8Array` (not `File`) because File objects cannot be reliably transferred via Comlink structured clone in all browsers. The main thread converts File to Uint8Array via `fetchFile` before calling the worker.

The worker returns `Blob[]` — the main thread creates `URL.createObjectURL()` for each blob. This keeps blob URL creation on the main thread where the URLs are consumed.

**Step 1: Create `src/hooks/useThumbnailExtractor.ts`**

This hook watches for new video clips and triggers thumbnail extraction. It tracks which clips have already been processed to avoid re-extraction on undo/redo (Research Pitfall 4).

```typescript
import { useEffect, useRef } from 'react'
import { useStore } from '../store'
import { getFfmpegProxy } from '../workers/ffmpeg.proxy'
import { computeTimestamps } from '../utils/computeTimestamps'
import { fetchFile } from '@ffmpeg/util'

export function useThumbnailExtractor() {
  // Track clip IDs that have been processed or are in-progress
  // This is a ref (outside store) so undo/redo doesn't re-trigger extraction
  const processedRef = useRef<Set<string>>(new Set())

  useEffect(() => {
    // Subscribe to store changes
    const unsubscribe = useStore.subscribe((state, prevState) => {
      // Find new video clips that need thumbnail extraction
      const newClipIds = state.tracks.video.clipIds.filter(
        (id) =>
          state.clips[id] &&
          state.clips[id].trackId === 'video' &&
          state.clips[id].thumbnailUrls.length === 0 &&
          !processedRef.current.has(id)
      )

      for (const clipId of newClipIds) {
        const clip = state.clips[clipId]
        if (!clip) continue

        processedRef.current.add(clipId)

        // Fire and forget — async extraction in worker
        ;(async () => {
          try {
            const proxy = getFfmpegProxy()
            const timestamps = computeTimestamps(clip.sourceDuration)
            const fileData = await fetchFile(clip.sourceFile)
            const blobs = await proxy.extractFrames(fileData, timestamps)

            // Convert blobs to object URLs on main thread
            const urls = (blobs as Blob[]).map((blob: Blob) =>
              URL.createObjectURL(blob)
            )

            // Update store with thumbnail URLs
            useStore.setState((prev) => {
              const existing = prev.clips[clipId]
              if (!existing) return prev // clip was deleted during extraction
              return {
                clips: {
                  ...prev.clips,
                  [clipId]: { ...existing, thumbnailUrls: urls },
                },
              }
            })
          } catch (err) {
            console.error(`Thumbnail extraction failed for clip ${clipId}:`, err)
            // Don't remove from processedRef — don't retry on error
          }
        })()
      }
    })

    return unsubscribe
  }, [])
}
```

Key implementation details:
- `processedRef` (a Set outside the store) prevents re-extraction on undo/redo (Research Pitfall 4)
- Only video clips get thumbnails (audio clips have no visual frames)
- `fetchFile` converts File to Uint8Array on the main thread before sending to worker
- Blob URLs created on main thread from worker-returned Blobs
- `setState` used directly (not an action) for thumbnail URL update — this still goes through Zundo but that's acceptable since thumbnail data is part of `clips` (tracked state). If this causes undo issues, the thumbnailUrls update could be moved to a non-tracked path, but for v1 simplicity this approach is fine.
- Fire-and-forget async pattern — no await needed at the call site

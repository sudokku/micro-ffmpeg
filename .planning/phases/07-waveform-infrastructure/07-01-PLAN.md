   ```typescript
   /**
    * Downsample an AudioBuffer's first channel to `peakCount` amplitude peaks.
    * Returns a peak-normalized number[] where the loudest peak is 1.0.
    */
   export function extractPeaks(audioBuffer: AudioBuffer, peakCount: number): number[] {
     const channelData = audioBuffer.getChannelData(0)
     const blockSize = Math.floor(channelData.length / peakCount)
     if (blockSize === 0) return new Array(peakCount).fill(0)

     const peaks: number[] = []

     for (let i = 0; i < peakCount; i++) {
       const start = i * blockSize
       let max = 0
       for (let j = start; j < start + blockSize; j++) {
         const abs = Math.abs(channelData[j])
         if (abs > max) max = abs
       }
       peaks.push(max)
     }

     // Peak-normalize: divide by the highest value so bars use full clip height
     const globalMax = Math.max(...peaks)
     if (globalMax === 0) return peaks // all silence — avoid divide-by-zero
     return peaks.map((p) => p / globalMax)
   }
   ```

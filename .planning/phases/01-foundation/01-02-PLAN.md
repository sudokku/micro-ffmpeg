    1. Create `src/store/index.ts`:

       ```typescript
       import { create } from 'zustand'
       import { temporal } from 'zundo'
       import type { StoreState, TrackedState } from './types'

       export const useStore = create<StoreState>()(
         temporal(
           (_set) => ({
             tracks: {
               video: { id: 'video', clipIds: [] },
               audio: { id: 'audio', clipIds: [] },
             },
             clips: {},
             clipSettings: {},
             ui: { selectedClipId: null, activeTool: 'select' },
             export: { status: 'idle', progress: 0 },
           }),
           {
             partialize: (state): TrackedState => {
               const { ui, export: _export, ...tracked } = state
               return tracked
             },
           },
         ),
       )
       ```

       CRITICAL: The `partialize` function MUST destructure out `ui` and `export` and return only `tracks`, `clips`, `clipSettings`. This is a locked decision — the previous project's #1 bug was including ui/export in undo history.

    2. Create `src/hooks/useTemporalStore.ts`:

       ```typescript
       import type { TemporalState } from 'zundo'
       import { useStore } from '../store'
       import type { TrackedState } from '../store/types'

       export const useTemporalStore = <T,>(
         selector: (state: TemporalState<TrackedState>) => T,
       ): T => useStore.temporal(selector)
       ```
